## Usage

```tsx
import { Avatar, Profile } from '@fidely-ui/react'
```

```tsx
<Profile.Root>
  <Profile.AvatarWrapper>
    <Avatar.Root>
      <Avatar.Fallback />
      <Avatar.Image />
    </Avatar.Root>
  </Profile.AvatarWrapper>

  <Profile.Details>
    <Profile.Name />
    <Profile.Title />
  </Profile.Details>
</Profile.Root>
```

## Props

| Prop          | Type                         | Default        | Description                                                |
| ------------- | ---------------------------- | -------------- | ---------------------------------------------------------- |
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Defines the layout direction of the **Profile** component. |
